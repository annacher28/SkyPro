string.index("") всегда возвращает 0 для любой строки 
Пустой символ — это некорректный ввод

Правильный вид функции 
def contains(self, string: str, symbol: str) -> bool:
    if symbol == "":
        return False
    try:
        return string.index(symbol) > -1
    except ValueError:
        return False
